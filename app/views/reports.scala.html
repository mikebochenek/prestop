@(restaurants: Seq[Restaurant], dishes: Seq[Dish], deletedDishes: Seq[Dish], users: Seq[UserFull], friends: Seq[Friend], reservations: Seq[Reservation], activities: Seq[ActivityLog], recent10: Seq[ActivityLog], imagesCount: Long, adminStats: String)(implicit session: Session, lang: Lang) @main(title = "reports") { @if(session.get("usertype").get.toString.equals("7")) {

@Messages("reports.restaurants") (@restaurants.size)

@for((restaurant, index) <- restaurants.zipWithIndex) { }
@index @restaurant.id @restaurant.name

@Messages("reports.dishes") (@dishes.size)

@for((dish, index) <- dishes.zipWithIndex) { }
# @Messages("reports.id") @Messages("reports.lastupdate") @Messages("reports.name") @Messages("reports.price") @Messages("reports.restaurant")
@index @dish.id @dish.lastupdate.format("dd MMM YYYY HH:mm") @dish.name @("%.2f".format(dish.price)) @dish.restaurant_id

@Messages("reports.deleted_dishes") (@deletedDishes.size)

@for((dish, index) <- deletedDishes.zipWithIndex) { }
# @Messages("reports.id") @Messages("reports.lastupdate") @Messages("reports.name") @Messages("reports.price") @Messages("reports.restaurant")
@index @dish.id @dish.lastupdate.format("dd MMM YYYY HH:mm") @dish.name @("%.2f".format(dish.price)) @dish.restaurant_id

@Messages("reports.users") (@users.size)

@for((user, index) <- users.zipWithIndex) { }
# @Messages("reports.id") @Messages("reports.email") @Messages("reports.username") @Messages("reports.createdate")
@index @user.id @user.email @user.username @user.createdate.format("dd MMM YYYY HH:mm")

@Messages("reports.friends") (@friends.size)

@for((friend, index) <- friends.zipWithIndex) { }
# @Messages("reports.id") user_id friend_user_id @Messages("reports.lastupdate")
@index @friend.id @friend.user_id @friend.friend_user_id @friend.lastupdate.format("dd MMM YYYY HH:mm")

Reservations (@reservations.size)

@for((reservation, index) <- reservations.zipWithIndex) { }
@index @reservation.user_id @reservation.restaurant_id @reservation.id @reservation.reservationtime.format("dd MMM YYYY HH:mm") @reservation.guestcount @reservation.special_requests @reservation.lastupdate.format("dd MMM YYYY HH:mm") @reservation.status

Activities / Likes (@activities.size)

(here we don't show the values for activity_type == 7, because there are too many)
@for((activity, index) <- activities.filter{x => x.activity_type != 7}.zipWithIndex) { }
# id user_id activity_type (11=like) activity_subtype (dish_id) activity_details lastupdate
@index @activity.id @activity.user_id @activity.activity_type @activity.activity_subtype @activity.activity_details @activity.createdate.format("dd MMM YYYY HH:mm")

Ten Most Recent Activities (@recent10.size)

@for((activity, index) <- recent10.zipWithIndex) { }
# id user_id activity_type (11=like, 7=recommend) activity_subtype (dish_id) activity_details lastupdate
@index @activity.id @activity.user_id @activity.activity_type @activity.activity_subtype @activity.activity_details.toString.take(60) @activity.createdate.format("dd MMM YYYY HH:mm")

Images (@imagesCount)

Tags (@Tag.countAll)

Tags References (@TagRef.countAll)

System Info

@Html(adminStats)
} }
