@(restaurants: Seq[Restaurant], dishes: Seq[Dish], users: Seq[User], friends: Seq[Friend], reservations: Seq[Reservation], activities: Seq[ActivityLog], images: Seq[Image], adminStats: String)(implicit session: Session, lang: Lang) @main(title = "reports") {

Restaurants

@for((restaurant, index) <- restaurants.zipWithIndex) { }
@index @restaurant.id @restaurant.lastupdate.format("dd MMM YYYY HH:MM") @restaurant.name

Dishes

@for((dish, index) <- dishes.zipWithIndex) { }
@index @dish.id @dish.lastupdate.format("dd MMM YYYY HH:MM") @dish.name @("%.2f".format(dish.price))

Users

@for((user, index) <- users.zipWithIndex) { }
@index @user.id @user.email @user.username

Friends

@for((friend, index) <- friends.zipWithIndex) { }
@index @friend.id @friend.friend_user_id @friend.user_id @friend.lastupdate.format("dd MMM YYYY HH:MM")

Reservations

@for((reservation, index) <- reservations.zipWithIndex) { }
@index @reservation.user_id @reservation.restaurant_id @reservation.id @reservation.reservationtime.format("dd MMM YYYY HH:MM") @reservation.guestcount @reservation.special_requests @reservation.lastupdate.format("dd MMM YYYY HH:MM") @reservation.status

Activities

@for((activity, index) <- activities.zipWithIndex) { }
@index @activity.id @activity.user_id @activity.activity_type @activity.activity_subtype @activity.activity_details @activity.createdate.format("dd MMM YYYY HH:MM")

Images

@for((image, index) <- images.zipWithIndex) { }
@index @image.id @image.filename @image.url @image.restaurant_id @image.dish_id @image.lastupdate.format("dd MMM YYYY HH:MM")

System Info

@Html(adminStats)
}
